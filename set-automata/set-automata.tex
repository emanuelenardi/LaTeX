% https://latex.org/forum/viewtopic.php?t=22367
\DeclareRobustCommand\circled[1]{%
	\tikz[baseline = (char.base)]{%
		\node[draw, circle, inner sep = 1pt]%
			(char) {#1};%
	}%
}
